Fontawesome

https://fontawesome.com/

Utilisation

Via les sprites : \#FA_ICON{name,class,alt}
#FA_ICON{beer,'',Beer} \#FA_ICON{beer,'',Beer}
#FA_ICON{brands#ubuntu,icon-sm,Ubuntu} \#FA_ICON{brands#ubuntu,icon-sm,Ubuntu}
#FA_ICON{regular#kiss,icon-lg,Kiss} \#FA_ICON{regular#kiss,icon-lg,Kiss}

En utilisant les SVG un par un selon le besoin :
#FA_ICON{img/fa/solid/beer.svg,'',Beer} \#FA_ICON{img/fa/solid/beer.svg,'',Beer}
#FA_ICON{img/fa/brands/ubuntu.svg,icon-sm,Ubuntu} \#FA_ICON{img/fa/brands/ubuntu.svg,icon-sm,Ubuntu}
#FA_ICON{img/fa/regular/kiss.svg,icon-lg,Kiss} \#FA_ICON{img/fa/regular/kiss.svg,icon-lg,Kiss}

Solid img/fa/solid.svg [(#CHEMIN{img/fa/solid.svg}|filesize|taille_en_octets)]

#SET{icons,#CHEMIN{img/fa/solid.svg}|lister_icones_svg}

#SET{name,#VALEUR} #FA_ICON{#GET{name},'',#GET{name}}

Regular img/fa/regular.svg [(#CHEMIN{img/fa/regular.svg}|filesize|taille_en_octets)]

#SET{icons,#CHEMIN{img/fa/regular.svg}|lister_icones_svg}

#SET{name,#VALEUR} #FA_ICON{regular\##GET{name},'',regular\##GET{name}}

Brands img/fa/brands.svg [(#CHEMIN{img/fa/brands.svg}|filesize|taille_en_octets)]

#SET{icons,#CHEMIN{img/fa/brands.svg}|lister_icones_svg}

#SET{name,#VALEUR} #FA_ICON{brands\##GET{name},'',brands\##GET{name}}

#SET{icons,#CHEMIN{img/fa/regular.svg}|lister_icones_svg}

Large .icon-lg

#SET{name,#VALEUR} #FA_ICON{regular\##GET{name},'icon-lg',regular\##GET{name}}

Small .icon-sm

#SET{name,#VALEUR} #FA_ICON{regular\##GET{name},'icon-sm',regular\##GET{name}}
